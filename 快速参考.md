## 一、数据集模块接口

### 1.1 BaseDataset

```python
class BaseDataset(torch.utils.data.Dataset):
    """基类：统一的数据处理接口"""
    
    def __init__(
        self,
        mode: str,              # 'train', 'val', 'test'
        shape: List[int] = None,        # [512, 1024] 输入大小
        max_wall_num: int = 999,        # 最大墙数过滤
        aug: Dict = None,               # 增强参数
        camera_height: float = 1.6,     # 相机高度(米)
        patch_num: int = 256,           # 深度采样点数
        keys: List[str] = None          # 输出键列表
    ) → None
    
    def __len__(self) → int
    
    def __getitem__(self, idx: int) → Dict[str, np.ndarray]
        """
        返回:
            {
                'image': np.ndarray(3, 512, 1024),        # 通道优先
                'depth': np.ndarray(256,),                # float32
                'ratio': float,                           # 房间高度比例
                'new_depth': np.ndarray(256,),            # float32 (可选)
                'opening': np.ndarray(1024,),             # float32 (可选)
                'corner_heat_map': np.ndarray(256,),      # float32 (可选)
                'id': str,                                # 样本ID
                'corners': np.ndarray(128, 2),            # 所有角点(可选)
            }
        """
    
    @staticmethod
    def get_depth(
        corners: np.ndarray,        # (N, 2) UV坐标
        plan_y: float = 1,          # 计划高度
        length: int = 256,          # 采样点数
        visible: bool = True        # 是否仅可见部分
    ) → np.ndarray  # (length,) 深度值
    
    def process_data(
        self,
        label: Dict,                # 标签字典
        image: np.ndarray,          # (H, W, 3) 图像
        patch_num: int,             # 采样点数
        new_label: Dict = None      # 新标签(可选)
    ) → Dict[str, np.ndarray]
```

## 二、模型模块接口

### 2.1 Bi_Layout 主模型

```python
class Bi_Layout(BaseModule):
    """双头全景布局识别模型"""
    
    def __init__(
        self,
        ckpt_dir: str = None,                    # 检查点目录
        backbone: str = 'resnet50',              # 特征提取器
        dropout: float = 0.0,
        output_name: str = 'Bi_Layout',
        decoder_name: str = 'Share_Feature_Guidance_Module',  # Transformer解码器
        win_size: int = 8,                       # 窗口大小
        depth: int = 6,                          # Transformer层数
        output_number: int = 2,                  # 输出头数(1或2)
        feature_channel: int = 1024,             # 特征维度
        height_compression_scale: int = 8,       # 高度压缩比
        embedding_channel: int = 512,            # 嵌入维度
        use_same_head: bool = False,             # 两头共享参数
        share_TF: bool = True,                   # 共享Transformer
        two_conv_out: bool = False,
        ape: str = None,                         # 绝对位置编码
        rpe: str = None,                         # 相对位置编码
        corner_heat_map: bool = False,
        rpe_pos: int = 1
    ) → None
    
    def forward(self, x: torch.Tensor) → Dict[str, torch.Tensor]
        """
        输入:
            x: (B, 3, 512, 1024)  全景图像
        
        返回:
            {
                'depth': (B, 256),      # 原始布局深度
                'ratio': (B, 1),        # 房间高度比例
                'new_depth': (B, 256),  # 新布局深度(output_number=2时)
                'corner_heat_map': (B, 256),  # 可选
            }
        """
    
    def bi_layout_outputs(
        self,
        x: torch.Tensor,            # (B, 256, 1024)
        new_x: torch.Tensor         # (B, 256, 1024)
    ) → Dict[str, torch.Tensor]
```

```python
    def forward(self, gt: torch.Tensor, pred: torch.Tensor) → torch.Tensor

class GradLoss(nn.Module):
    def forward(self, gt: Dict, pred: Dict) → torch.Tensor or List[torch.Tensor]
```

## 九、主要命令行接口

### 9.1 main.py

```bash
python main.py \
  --cfg config/mp3d.yaml          # 配置文件路径
  --mode train|val|test           # 运行模式
  --bs 16                         # 批大小(覆盖配置)
  --save_eval                     # 保存评估结果
  --post_processing manhattan     # 后处理方法
  --need_cpe                      # 计算角点误差
  --need_f1                       # 计算F1
  --need_rmse                     # 计算RMSE
  --force_cube                    # 强制立方体
  --ckpt_option best              # best/last/average

# 示例:
python main.py --cfg src/config/mp3d.yaml --mode test --save_eval
```

### 9.2 inference.py

```bash
python inference.py \
  --img_glob "path/to/images/*.png"    # 输入图像路径
  --cfg config/mp3d.yaml               # 配置文件
  --output_dir src/output              # 输出目录
  --post_processing manhattan          # 后处理
```
